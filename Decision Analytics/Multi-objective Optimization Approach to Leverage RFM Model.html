Calculate RFM¶

Determine which customers are high-value customers by examining how recently a customer has purchased (recency), how often they purchase (frequency), and how much the customer spends (monetary)

In [40]:
# Calculate revenue per transaction
df_train['MonetaryValue'] = df_train['Quantity'] * df_train['Price']

# Calculate recency
max_date = max(df_train['InvoiceDate'])
df_train['Recency'] = (max_date - df_train['InvoiceDate'])

# Prepare the Recency, Frequency, MonetaryValue columns
df_train_rfm = df_train.groupby(['Customer ID']).agg({
    'Recency': min,
    'Invoice': 'count',
    'MonetaryValue': sum
}).reset_index()
df_train_rfm.columns = ['CustomerID','Recency', 'Frequency', 'MonetaryValue']

    {key: (purchase_prob[key] * expected_revenue[key]  - avg_cost) * customer_count[key] if key[0] in (1, 2) else 0
    for key in rfm_segment}),
    0,  weight=200, name='prioritize_low_r')

    {key: (purchase_prob[key] * expected_revenue[key] - avg_cost) * customer_count[key] if key[1] in (3, 4) else 0
    for key in rfm_segment}),
    1,  weight=100, name='prioritize_high_f')

    {key: (purchase_prob[key] * expected_revenue[key] - avg_cost) * customer_count[key] if key[2] in (3, 4) else 0

)

        decision = m.getAttr('x')
        print('Objective value: ', (df['CustomerCount'] * df['ExpectedRevenue'] * df['Probability']) @ decision)
        m.printAttr(['x'])
        m.printAttr(['Sense', 'Slack', 'RHS'])
    else:
        print('No solution')
